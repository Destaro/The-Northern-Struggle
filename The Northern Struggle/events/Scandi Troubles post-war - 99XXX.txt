# DEN wins in Britain (about NTU)
country_event = {
	id = 99032
	title = "REV32.T"
	desc = "REV32.D"
	
	trigger = {
		tag = DEN
		NOT = { war_with = ENG }
		has_global_flag = eng_revolt
		has_global_flag = ntu_loyal
		OR = {
			is_our_vassal = ENG
			NOT = { exists = ENG }
		}
	}
	fire_only_once = yes
	
	mean_time_to_happen = { days = 1 }
	
	option = { # Give stuff to NTU
		name = "REV32.A1"
		
		any_owned = {
			limit = {
				is_core = ENG
			}
			secede_province = NTU
			remove_core = DEN
			add_core = NTU
		}
		badboy = -1
		prestige = 5
		clr_global_flag = eng_revolt
		ai_chance = { factor = 100 }
	}
	
	option = { # Keep everything
		name = "REV32.A2"
		
		relation = {
			who = NTU
			value = -200
		}
		any_pop = {
			limit = {
				culture = Northumbriam
			}
			militancy = 2
			consciousness = 2
		}
		badboy = 2
		prestige = -10
		clr_global_flag = eng_revolt
		ai_chance = { factor = 0 }
	}
}
