	title = "REV12.T"
	desc = "REV12.D"
	
	major = yes
	
	trigger = {
		has_country_flag = riots_suppressed
		tag = DEN
		OR = {
			has_country_flag = increased_taxes
			has_country_flag = increased_conscription
		}
	}
	fire_only_once = yes

	option = { # Play Denmark
		name = "REV12.A1"
		
		any_owned = {
			limit = {
		        OR = {
					region = ENG_300

					region = ENG_284
					region = ENG_291
		        }
				NOT = { province_id = 290 }

			}

		}
		release_vassal = ENG
		casus_belli = {
			target = ENG
			type = conquest
			months = 12
		}
		war = {
			target = ENG
			attacker_goal = {
				casus_belli = conquest
			}
			defender_goal = {
					casus_belli = acquire_core_state
					state_province_id = 284
			}
			defender_goal = {
				casus_belli = acquire_core_state
				state_province_id = 292
			}
		}
		prestige = -50
		set_global_flag = eng_revolt
